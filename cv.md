function findMissingLetter(arr) {
  let start = arr[0].charCodeAt(0);
  
  for (let i = 1; i < arr.length; i++) {
    if (arr[i].charCodeAt(0) - start > 1) {
      return String.fromCharCode(start + 1);
    }
    
    start = arr[i].charCodeAt(0);
  }
  
  return undefined;
}

const letters = ['A', 'B', 'C', 'E', 'F', 'G'];
